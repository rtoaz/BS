-- 检测设备类型
local function detectDeviceType()
    local screenSize = workspace.CurrentCamera.ViewportSize
    if screenSize.X < 600 then
        return "Mobile"
    elseif screenSize.X < 1024 then
        return "Tablet"
    else
        return "Desktop"
    end
end

-- 动画工具函数
local function Tween(obj, info, props)
    local tween = TweenService:Create(obj, info or TweenInfo.new(0.3, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), props)
    tween:Play()
    return tween
end

local function TweenFast(obj, props)
    return Tween(obj, TweenInfo.new(0.2, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), props)
end

local function TweenSlow(obj, props)
    return Tween(obj, TweenInfo.new(0.5, Enum.EasingStyle.Quart, Enum.EasingDirection.Out), props)
end

-- 涟漪效果
local function Ripple(obj, x, y)
    task.spawn(function()
        if obj.ClipsDescendants ~= true then
            obj.ClipsDescendants = true
        end
        
        local ripple = Instance.new("Frame")
        ripple.Name = "Ripple"
        ripple.Parent = obj
        ripple.BackgroundColor3 = Theme.TextPrimary
        ripple.BackgroundTransparency = 0.8
        ripple.BorderSizePixel = 0
        ripple.ZIndex = obj.ZIndex + 10
        
        local corner = Instance.new("UICorner")
        corner.CornerRadius = UDim.new(1, 0)
        corner.Parent = ripple
        
        local absSize = obj.AbsoluteSize
        local size = math.max(absSize.X, absSize.Y) * 1.5
        
        local posX = x or absSize.X / 2
        local posY = y or absSize.Y / 2
        
        ripple.Position = UDim2.new(0, posX - size/2, 0, posY - size/2)
        ripple.Size = UDim2.new(0, 0, 0, 0)
        
        Tween(ripple, TweenInfo.new(0.5, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {
            Size = UDim2.new(0, size, 0, size),
            BackgroundTransparency = 1
        })
        
        task.wait(0.5)
        ripple:Destroy()
    end)
end

-- 渐变背景
local function AddGradient(parent, color1, color2, rotation)
    local gradient = Instance.new("UIGradient")
    gradient.Color = ColorSequence.new({
        ColorSequenceKeypoint.new(0, color1 or Theme.GradientStart),
        ColorSequenceKeypoint.new(1, color2 or Theme.GradientEnd)
    })
    gradient.Rotation = rotation or 45
    gradient.Parent = parent
    return gradient

-- 玻璃效果
local function ApplyGlassEffect(frame, transparency)
    frame.BackgroundColor3 = Theme.GlassBackground
    frame.BackgroundTransparency = transparency or Theme.GlassTransparency
    
    local stroke = Instance.new("UIStroke")
    stroke.Color = Theme.Border
    stroke.Transparency = 0.6
    stroke.Thickness = 1
    stroke.Parent = frame
    
    local corner = frame:FindFirstChildOfClass("UICorner")
    if not corner then
        corner = Instance.new("UICorner")
        corner.CornerRadius = UDim.new(0, 8)
        corner.Parent = frame
    end
    
    return stroke
end

-- 标签切换状态
local switchingTabs = false
local function switchTab(newTab)
    if switchingTabs or not newTab then return end
    
    local oldTab = library.currentTab
    if oldTab and oldTab[1] == newTab[1] then return end
    
    switchingTabs = true
    library.currentTab = newTab
    
    -- 淡出旧标签页
    if oldTab then
        TweenFast(oldTab[1], { ImageTransparency = 0.4, ImageColor3 = Theme.TextMuted })
        TweenFast(oldTab[1]:FindFirstChild("TabText"), { TextTransparency = 0.4, TextColor3 = Theme.TextMuted })
        TweenFast(oldTab[2], { Position = UDim2.new(0.02, 0, 0, 0), Size = UDim2.new(0.96, 0, 0.95, 0) })
        
        task.delay(0.15, function()
            oldTab[2].Visible = false
        end)
    end
    
    -- 显示并动画新标签页
    newTab[2].Visible = true
    newTab[2].Position = UDim2.new(0.04, 0, 0, 0)
    newTab[2].Size = UDim2.new(0.92, 0, 0.9, 0)
    
    TweenFast(newTab[2], { Position = UDim2.new(0, 0, 0, 0), Size = UDim2.new(1, 0, 1, 0) })
    TweenFast(newTab[1], { ImageTransparency = 0, ImageColor3 = Theme.Primary })
    TweenFast(newTab[1]:FindFirstChild("TabText"), { TextTransparency = 0, TextColor3 = Theme.TextPrimary })
    
    task.delay(0.2, function()
        switchingTabs = false
    end)
end

-- 拖拽功能
local function MakeDraggable(frame, handle)
    handle = handle or frame
    local dragging = false
    local dragInput, dragStart, startPos
    
    handle.InputBegan:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1 or 
           input.UserInputType == Enum.UserInputType.Touch then
            dragging = true
            dragStart = input.Position
            startPos = frame.Position
            
            input.Changed:Connect(function()
                if input.UserInputState == Enum.UserInputState.End then
                    dragging = false
                end
            end)
        end
    end)
    
    handle.InputChanged:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseMovement or 
           input.UserInputType == Enum.UserInputType.Touch then
            dragInput = input
        end
    end)
    
    UserInputService.InputChanged:Connect(function(input)
        if input == dragInput and dragging then
            local delta = input.Position - dragStart
            TweenFast(frame, {
                Position = UDim2.new(
                    startPos.X.Scale, 
                    startPos.X.Offset + delta.X, 
                    startPos.Y.Scale, 
                    startPos.Y.Offset + delta.Y
                )
            })
        end
    end)
end

            if Theme[k] then
                Theme[k] = v

                assert(default, "No default key provided")
                
                local defaultKey = typeof(default) == "string" and Enum.KeyCode[default] or default
                local banned = {
                    Return = true, Space = true, Tab = true,
                    Backquote = true, CapsLock = true, Escape = true, Unknown = true
                }
                local shortNames = {
                    RightControl = "R Ctrl", LeftControl = "L Ctrl",
                    LeftShift = "L Shift", RightShift = "R Shift",
                    Semicolon = ";", Quote = '"',
                    LeftBracket = "[", RightBracket = "]",
                    Equals = "=", Minus = "-",
                    RightAlt = "R Alt", LeftAlt = "L Alt"
                }
                
                local bindKey = defaultKey
                local keyText = shortNames[defaultKey.Name] or defaultKey.Name
                
                local keybindFrame = Instance.new("Frame")
                keybindFrame.Name = text .. "_Keybind"
                keybindFrame.Parent = content

                local keyBtn = Instance.new("TextButton")
                keyBtn.Name = "KeyBtn"
                keyBtn.Parent = keybindFrame
                keyBtn.BackgroundColor3 = Theme.BackgroundSecondary
                keyBtn.BorderSizePixel = 0
                keyBtn.Position = UDim2.new(1, -90, 0.5, -12)
                keyBtn.Size = UDim2.new(0, 80, 0, 24)
                keyBtn.Font = Enum.Font.GothamBold
                keyBtn.Text = keyText
                keyBtn.TextColor3 = Theme.TextSecondary
                keyBtn.TextSize = 12
                keyBtn.AutoButtonColor = false
                
                local keyBtnCorner = Instance.new("UICorner")
                keyBtnCorner.CornerRadius = UDim.new(0, 6)
                keyBtnCorner.Parent = keyBtn
                
                local isListening = false
                
                keyBtn.MouseButton1Click:Connect(function()
                    Ripple(keyBtn)
                    isListening = true
                    keyBtn.Text = "..."
                    keyBtn.TextColor3 = Theme.Primary
                end)
                
                UserInputService.InputBegan:Connect(function(input, gameProcessed)
                    if gameProcessed then return end
                    
                    if isListening then
                        if input.UserInputType == Enum.UserInputType.Keyboard then
                            local keyName = input.KeyCode.Name
                            if banned[keyName] then return end
                            
                            bindKey = input.KeyCode
                            keyText = shortNames[keyName] or keyName
                            keyBtn.Text = keyText
                            keyBtn.TextColor3 = Theme.TextSecondary
                            isListening = false
                        end
                    elseif input.KeyCode == bindKey then
                        callback(bindKey.Name)
                    end
                end)
                
                return keybindFrame
            end

                addBtn.TextColor3 = Theme.TextSecondary

                local function setValue(value)
                    value = math.clamp(value, min, max)
                    if precise then
                        value = tonumber(string.format("%.1f", value))
                    else
                        value = math.floor(value)
                    end
                    
                    library.flags[flag] = value
                    valueBox.Text = tostring(value)
                    
                    local percent = (value - min) / (max - min)
                    TweenFast(sliderFill, { Size = UDim2.new(percent, 0, 1, 0) })
                    
                    callback(value)
                end
                
                -- 拖动功能
                local dragging = false
                
                sliderTrack.InputBegan:Connect(function(input)
                    if input.UserInputType == Enum.UserInputType.MouseButton1 or 
                       input.UserInputType == Enum.UserInputType.Touch then
                        dragging = true
                        local percent = math.clamp((input.Position.X - sliderTrack.AbsolutePosition.X) / sliderTrack.AbsoluteSize.X, 0, 1)
                        setValue(min + (max - min) * percent)
                    end
                end)
                
                UserInputService.InputEnded:Connect(function(input)
                    if input.UserInputType == Enum.UserInputType.MouseButton1 or 
                       input.UserInputType == Enum.UserInputType.Touch then
                        dragging = false
                    end
                end)
                
                UserInputService.InputChanged:Connect(function(input)
                    if dragging and (input.UserInputType == Enum.UserInputType.MouseMovement or 
                                     input.UserInputType == Enum.UserInputType.Touch) then
                        local percent = math.clamp((input.Position.X - sliderTrack.AbsolutePosition.X) / sliderTrack.AbsoluteSize.X, 0, 1)
                        setValue(min + (max - min) * percent)
                    end
                end)
                
                minusBtn.MouseButton1Click:Connect(function()
                    setValue(library.flags[flag] - (precise and 0.1 or 1))
                end)
                
                addBtn.MouseButton1Click:Connect(function()
                    setValue(library.flags[flag] + (precise and 0.1 or 1))
                end)
                
                valueBox.FocusLost:Connect(function()
                    local num = tonumber(valueBox.Text)
                    if num then
                        setValue(num)
                    else
                        valueBox.Text = tostring(library.flags[flag])
                    end
                end)
                
                setValue(default)
                
                local funcs = {
                    SetValue = function(self, value)
                        setValue(value)
                    end,
                    GetValue = function(self)
                        return library.flags[flag]
                    end
                }
                
                return funcs
            end

                    ColorSequenceKeypoint.new(1, Color3.fromRGB(255, 0, 0))
                })
                hueGradient.Parent = hueSlider
                
                local hueCorner = Instance.new("UICorner")
                hueCorner.CornerRadius = UDim.new(0, 4)
                hueCorner.Parent = hueSlider
                
                local hueIndicator = Instance.new("Frame")
                hueIndicator.Name = "Indicator"
                hueIndicator.Parent = hueSlider
                hueIndicator.BackgroundColor3 = Theme.TextPrimary
                hueIndicator.BorderSizePixel = 0
                hueIndicator.Size = UDim2.new(0, 4, 1, 4)
                hueIndicator.Position = UDim2.new(0, -2, 0, -2)
                
                -- 饱和度和亮度区域
                local svFrame = Instance.new("Frame")
                svFrame.Name = "SVFrame"
                svFrame.Parent = pickerPanel
                svFrame.BackgroundColor3 = Color3.fromRGB(255, 255, 255)
                svFrame.BorderSizePixel = 0
                svFrame.Position = UDim2.new(0, 10, 0, 40)
                svFrame.Size = UDim2.new(0.6, 0, 0, 80)
                
                local svCorner = Instance.new("UICorner")
                svCorner.CornerRadius = UDim.new(0, 8)
                svCorner.Parent = svFrame
                
                local svIndicator = Instance.new("Frame")
                svIndicator.Name = "Indicator"
                svIndicator.Parent = svFrame
                svIndicator.BackgroundColor3 = Theme.TextPrimary
                svIndicator.BorderSizePixel = 0
                svIndicator.Size = UDim2.new(0, 8, 0, 8)
                svIndicator.Position = UDim2.new(0.5, -4, 0.5, -4)
                
                local svIndicatorCorner = Instance.new("UICorner")
                svIndicatorCorner.CornerRadius = UDim.new(1, 0)
                svIndicatorCorner.Parent = svIndicator
                
                -- RGB输入
                local rgbFrame = Instance.new("Frame")
                rgbFrame.Name = "RGBFrame"
                rgbFrame.Parent = pickerPanel
                rgbFrame.BackgroundTransparency = 1
                rgbFrame.Position = UDim2.new(0.65, 0, 0, 40)
                rgbFrame.Size = UDim2.new(0.35, -10, 0, 80)
                
                local rgbLayout = Instance.new("UIListLayout")
                rgbLayout.Parent = rgbFrame
                rgbLayout.SortOrder = Enum.SortOrder.LayoutOrder
                rgbLayout.Padding = UDim.new(0, 4)
                
                local rgbInputs = {}
                for _, colorName in ipairs({"R", "G", "B"}) do
                    local inputFrame = Instance.new("Frame")
                    inputFrame.Name = colorName .. "Frame"
                    inputFrame.Parent = rgbFrame
                    inputFrame.BackgroundColor3 = Theme.BackgroundSecondary
                    inputFrame.BorderSizePixel = 0
                    inputFrame.Size = UDim2.new(1, 0, 0, 22)
                    
                    local inputCorner = Instance.new("UICorner")
                    inputCorner.CornerRadius = UDim.new(0, 4)
                    inputCorner.Parent = inputFrame
                    
                    local label = Instance.new("TextLabel")
                    label.Name = "Label"
                    label.Parent = inputFrame
                    label.BackgroundTransparency = 1
                    label.Size = UDim2.new(0, 20, 1, 0)
                    label.Position = UDim2.new(0, 5, 0, 0)
                    label.Font = Enum.Font.GothamBold
                    label.Text = colorName
                    label.TextColor3 = colorName == "R" and Theme.Error or colorName == "G" and Theme.Success or Theme.Info
                    label.TextSize = 11
                    
                    local input = Instance.new("TextBox")
                    input.Name = "Input"
                    input.Parent = inputFrame
                    input.BackgroundTransparency = 1
                    input.Size = UDim2.new(1, -30, 1, 0)
                    input.Position = UDim2.new(0, 25, 0, 0)
                    input.Font = Enum.Font.Gotham
                    input.Text = "255"
                    input.TextColor3 = Theme.TextPrimary
                    input.TextSize = 11
                    
                    rgbInputs[colorName] = input
                end
                
                -- 当前颜色显示
                local currentColorFrame = Instance.new("Frame")
                currentColorFrame.Name = "CurrentColor"
                currentColorFrame.Parent = pickerPanel
                currentColorFrame.BackgroundColor3 = default
                currentColorFrame.BorderSizePixel = 0
                currentColorFrame.Position = UDim2.new(0, 10, 1, -30)
                currentColorFrame.Size = UDim2.new(1, -20, 0, 25)
                
                local currentColorCorner = Instance.new("UICorner")
                currentColorCorner.CornerRadius = UDim.new(0, 6)
                currentColorCorner.Parent = currentColorFrame
                
                local currentColorLabel = Instance.new("TextLabel")
                currentColorLabel.Name = "Label"
                currentColorLabel.Parent = currentColorFrame
                currentColorLabel.BackgroundTransparency = 1
                currentColorLabel.Size = UDim2.new(1, 0, 1, 0)
                currentColorLabel.Font = Enum.Font.GothamBold
                currentColorLabel.Text = "当前颜色"
                currentColorLabel.TextColor3 = Theme.TextPrimary
                currentColorLabel.TextSize = 12
                
                local h, s, v = default:ToHSV()
                
                local function updateColor()
                    local newColor = Color3.fromHSV(h, s, v)
                    colorPreview.BackgroundColor3 = newColor
                    currentColorFrame.BackgroundColor3 = newColor
                    svFrame.BackgroundColor3 = Color3.fromHSV(h, 1, 1)
                    
                    local r, g, b = math.floor(newColor.R * 255), math.floor(newColor.G * 255), math.floor(newColor.B * 255)
                    rgbInputs.R.Text = tostring(r)
                    rgbInputs.G.Text = tostring(g)
                    rgbInputs.B.Text = tostring(b)
                    
                    library.flags[flag] = newColor
                    callback(newColor)
                end
                
                local isOpen = false
                colorPreview.MouseButton1Click:Connect(function()
                    isOpen = not isOpen
                    pickerPanel.Visible = isOpen
                    if isOpen then
                        TweenFast(pickerPanel, { BackgroundTransparency = 0.1 })
                    end
                end)
                
                -- 色相拖动
                local hueDragging = false
                hueSlider.InputBegan:Connect(function(input)
                    if input.UserInputType == Enum.UserInputType.MouseButton1 then
                        hueDragging = true
                        local percent = math.clamp((input.Position.X - hueSlider.AbsolutePosition.X) / hueSlider.AbsoluteSize.X, 0, 1)
                        h = percent
                        hueIndicator.Position = UDim2.new(percent, -2, 0, -2)
                        updateColor()
                    end
                end)
                
                UserInputService.InputEnded:Connect(function(input)
                    if input.UserInputType == Enum.UserInputType.MouseButton1 then
                        hueDragging = false
                    end
                end)
                
                UserInputService.InputChanged:Connect(function(input)
                    if hueDragging and input.UserInputType == Enum.UserInputType.MouseMovement then
                        local percent = math.clamp((input.Position.X - hueSlider.AbsolutePosition.X) / hueSlider.AbsoluteSize.X, 0, 1)
                        h = percent
                        hueIndicator.Position = UDim2.new(percent, -2, 0, -2)
                        updateColor()
                    end
                end)
                
                -- SV拖动
                local svDragging = false
                svFrame.InputBegan:Connect(function(input)
                    if input.UserInputType == Enum.UserInputType.MouseButton1 then
                        svDragging = true
                        local x = math.clamp((input.Position.X - svFrame.AbsolutePosition.X) / svFrame.AbsoluteSize.X, 0, 1)
                        local y = math.clamp((input.Position.Y - svFrame.AbsolutePosition.Y) / svFrame.AbsoluteSize.Y, 0, 1)
                        s = x
                        v = 1 - y
                        svIndicator.Position = UDim2.new(x, -4, y, -4)
                        updateColor()
                    end
                end)
                
                UserInputService.InputChanged:Connect(function(input)
                    if svDragging and input.UserInputType == Enum.UserInputType.MouseMovement then
                        local x = math.clamp((input.Position.X - svFrame.AbsolutePosition.X) / svFrame.AbsoluteSize.X, 0, 1)
                        local y = math.clamp((input.Position.Y - svFrame.AbsolutePosition.Y) / svFrame.AbsoluteSize.Y, 0, 1)
                        s = x
                        v = 1 - y
                        svIndicator.Position = UDim2.new(x, -4, y, -4)
                        updateColor()
                    end
                end)
                
                -- RGB输入
                for colorName, input in pairs(rgbInputs) do
                    input.FocusLost:Connect(function()
                        local num = tonumber(input.Text)
                        if num then
                            num = math.clamp(math.floor(num), 0, 255)
                            local currentColor = library.flags[flag]
                            local r, g, b = currentColor.R * 255, currentColor.G * 255, currentColor.B * 255
                            if colorName == "R" then
                                r = num
                            elseif colorName == "G" then
                                g = num
                            else
                                b = num
                            end
                            local newColor = Color3.fromRGB(r, g, b)
                            h, s, v = newColor:ToHSV()
                            hueIndicator.Position = UDim2.new(h, -2, 0, -2)
                            svIndicator.Position = UDim2.new(s, -4, 1 - v, -4)
                            updateColor()
                        end
                    end)
                end
                
                updateColor()
                
                local funcs = {
                    SetColor = function(self, color)
                        h, s, v = color:ToHSV()
                        hueIndicator.Position = UDim2.new(h, -2, 0, -2)
                        svIndicator.Position = UDim2.new(s, -4, 1 - v, -4)
                        updateColor()
                    end,
                    GetColor = function(self)
                        return library.flags[flag]
                    end
                }
                
                return funcs
            end
            
            -- 进度条 (新功能)
            function section.ProgressBar(section, text, flag, default, min, max)
                min = min or 0
                max = max or 100
                default = default or 0
                assert(text, "No text provided")
                assert(flag, "No flag provided")
                library.flags[flag] = default
                
                local progressFrame = Instance.new("Frame")
                progressFrame.Name = text .. "_ProgressBar"
                progressFrame.Parent = content
                progressFrame.BackgroundColor3 = Theme.SurfaceLight
                progressFrame.BackgroundTransparency = 0.7
                progressFrame.BorderSizePixel = 0
                progressFrame.Size = UDim2.new(1, 0, 0, currentSize.ElementHeight)
                
                local progressCorner = Instance.new("UICorner")
                progressCorner.CornerRadius = UDim.new(0, 8)
                progressCorner.Parent = progressFrame
                
                local progressLabel = Instance.new("TextLabel")
                progressLabel.Name = "Label"
                progressLabel.Parent = progressFrame
                progressLabel.BackgroundTransparency = 1
                progressLabel.Size = UDim2.new(0.5, 0, 1, 0)
                progressLabel.Position = UDim2.new(0, 15, 0, 0)
                progressLabel.Font = Enum.Font.GothamBold
                progressLabel.Text = text
                progressLabel.TextColor3 = Theme.TextPrimary
                progressLabel.TextSize = currentSize.FontSize
                progressLabel.TextXAlignment = Enum.TextXAlignment.Left
                
                local valueLabel = Instance.new("TextLabel")
                valueLabel.Name = "Value"
                valueLabel.Parent = progressFrame
                valueLabel.BackgroundTransparency = 1
                valueLabel.Position = UDim2.new(0.5, 0, 0, 0)
                valueLabel.Size = UDim2.new(0.5, -15, 1, 0)
                valueLabel.Font = Enum.Font.GothamBold
                valueLabel.Text = default .. "/" .. max
                valueLabel.TextColor3 = Theme.TextSecondary
                valueLabel.TextSize = 12
                valueLabel.TextXAlignment = Enum.TextXAlignment.Right
                
                local barBg = Instance.new("Frame")
                barBg.Name = "BarBg"
                barBg.Parent = progressFrame
                barBg.BackgroundColor3 = Theme.BackgroundSecondary
                barBg.BorderSizePixel = 0
                barBg.Position = UDim2.new(0, 15, 0.7, -4)
                barBg.Size = UDim2.new(1, -30, 0, 8)
                
                local barBgCorner = Instance.new("UICorner")
                barBgCorner.CornerRadius = UDim.new(1, 0)
                barBgCorner.Parent = barBg
                
                local barFill = Instance.new("Frame")
                barFill.Name = "BarFill"
                barFill.Parent = barBg
                barFill.BackgroundColor3 = Theme.Primary
                barFill.BorderSizePixel = 0
                barFill.Size = UDim2.new((default - min) / (max - min), 0, 1, 0)
                
                local barGradient = AddGradient(barFill, Theme.GradientStart, Theme.GradientEnd, 90)
                
                local barFillCorner = Instance.new("UICorner")
                barFillCorner.CornerRadius = UDim.new(1, 0)
                barFillCorner.Parent = barFill
                
                local funcs = {
                    SetValue = function(self, value)
                        value = math.clamp(value, min, max)
                        library.flags[flag] = value
                        valueLabel.Text = value .. "/" .. max
                        TweenFast(barFill, { Size = UDim2.new((value - min) / (max - min), 0, 1, 0) })
                    end,
                    GetValue = function(self)
                        return library.flags[flag]
                    end,
                    SetMax = function(self, newMax)
                        max = newMax
                        valueLabel.Text = library.flags[flag] .. "/" .. max
                    end
                }
                
                return funcs
            end
            
            return section
        end
        
        return tab
    end
    
    return window
end

-- 主题设置函数
function library.SetTheme(newTheme)
    for k, v in pairs(newTheme) do
        if Theme[k] then
            Theme[k] = v
        end
    end
end

-- 获取当前主题
function library.GetTheme()
    return table.clone(Theme)
end

-- 获取设备类型
function library.GetDeviceType()
    return library.deviceType
end

-- 设置设备类型(用于测试)
function library.SetDeviceType(device)
    library.deviceType = device
    currentSize = Sizes[device]
end

return library
