function Tween(obj, t, data)
    if not obj or not obj.Parent then return end
    services.TweenService:Create(obj, TweenInfo.new(t[1], Enum.EasingStyle[t[2]], Enum.EasingDirection[t[3]]), data):Play()
    return true
end

function AnimeClick(obj)
    task.spawn(function()
        PlayClickSound()
        local originalSize = obj.Size
        Tween(obj, {0.1, "Quad", "Out"}, {Size = UDim2.new(originalSize.X.Scale * 0.9, originalSize.X.Offset * 0.9, originalSize.Y.Scale * 0.9, originalSize.Y.Offset * 0.9)})
        task.wait(0.1)
        Tween(obj, {0.3, "Elastic", "Out"}, {Size = originalSize})
    end)
end

function drag(frame, hold)
    if not hold then hold = frame end
    local dragging, dragInput, dragStart, startPos
    local function update(input)
        if not (frame and frame.Parent) then dragging = false; return end
        local delta = input.Position - dragStart
        frame.Position = UDim2.new(startPos.X.Scale, startPos.X.Offset + delta.X, startPos.Y.Scale, startPos.Y.Offset + delta.Y)
    end
    hold.InputBegan:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1 then
            dragging = true
            dragStart = input.Position
            startPos = frame.Position
            input.Changed:Connect(function()
                if input.UserInputState == Enum.UserInputState.End then dragging = false end
            end)
        end
    end)
    frame.InputChanged:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseMovement then dragInput = input end
    end)
    services.UserInputService.InputChanged:Connect(function(input)
        if input == dragInput and dragging then update(input) end
    end)
end

local switchingTabs = false

local function resetCurrentTabVisibility()
    local currentTab = library.currentTab
    if not currentTab then return end
    local tabFrame = currentTab[2]
    if not tabFrame then return end
    for _, section in ipairs(tabFrame:GetChildren()) do
        if section:IsA("Frame") and section.Name == "Section" then
            pcall(function() section.Visible = true end)
            local objs = section:FindFirstChild("Objs")
            if objs then
                for _, element in ipairs(objs:GetChildren()) do
                    if element:IsA("Frame") then
                        pcall(function() element.Visible = true end)
                    end
                end
            end
        end
    end
end

function switchTab(new)
    if switchingTabs then return end
    local old = library.currentTab
    if old == nil then
        if new and new[2] then
            new[2].Visible = true
        end
        library.currentTab = new
        if new and new[1] and new[1].TabText then
            Tween(new[1], {0.12, "Quad", "Out"}, {ImageTransparency = 0})
            Tween(new[1].TabText, {0.12, "Quad", "Out"}, {TextTransparency = 0})
        end
        resetCurrentTabVisibility()
        if SearchBox then SearchBox.Text = "" end
        return
    end
    if old[1] == new[1] then return end
    switchingTabs = true
    library.currentTab = new
    if old and old[1] and old[1].TabText then
        Tween(old[1], {0.12, "Quad", "Out"}, {ImageTransparency = 0.2})
        Tween(old[1].TabText, {0.12, "Quad", "Out"}, {TextTransparency = 0.2})
    end
    if new and new[1] and new[1].TabText then
        Tween(new[1], {0.12, "Quad", "Out"}, {ImageTransparency = 0})
        Tween(new[1].TabText, {0.12, "Quad", "Out"}, {TextTransparency = 0})
    end
    if old and old[2] then old[2].Visible = false end
    if new and new[2] then new[2].Visible = true end
    resetCurrentTabVisibility()
    if SearchBox then SearchBox.Text = "" end
    task.wait(0.12)
    switchingTabs = false
end

    local function filterCurrentTab(searchText)
        local currentTab = library.currentTab
        if not currentTab then return end
        local tabFrame = currentTab[2]
        if not tabFrame then return end
        searchText = searchText or ""
        local trimmed = searchText:gsub("^%s+", ""):gsub("%s+$", ""):lower()
        local isEmpty = trimmed == ""
        for _, section in ipairs(tabFrame:GetChildren()) do
            if section:IsA("Frame") and section.Name == "Section" then
                local sectionTitle = ""
                local sectionTextObj = section:FindFirstChild("SectionText")
                if sectionTextObj and sectionTextObj:IsA("TextLabel") then
                    sectionTitle = sectionTextObj.Text or ""
                end
                local sectionMatch = sectionTitle:lower():find(trimmed, 1, true) ~= nil
                local objs = section:FindFirstChild("Objs")
                if objs then
                    local anyVisible = false
                    for _, element in ipairs(objs:GetChildren()) do
                        if element:IsA("Frame") then
                            local elementText = ""
                            local btn = element:FindFirstChildOfClass("TextButton")
                            if btn then
                                elementText = btn.Text or ""
                            else
                                local lbl = element:FindFirstChildOfClass("TextLabel")
                                if lbl then
                                    elementText = lbl.Text or ""
                                else
                                    local txt = element:FindFirstChildOfClass("TextBox")
                                    if txt then
                                        elementText = txt.Text or ""
                                    end
                                end
                            end
                            local match = isEmpty or sectionMatch or (elementText:lower():find(trimmed, 1, true) ~= nil)
                            pcall(function() element.Visible = match end)
                            if match then anyVisible = true end
                        end
                    end
                    pcall(function() section.Visible = isEmpty or sectionMatch or anyVisible end)
                else
                    pcall(function() section.Visible = isEmpty or sectionMatch end)
                end
            end
        end
    end

            rotationConnection = services.RunService.RenderStepped:Connect(function(dt)
                if not isResizing then
                    rotationConnection:Disconnect()
                    rotationConnection = nil
                    return
                end
                rotationAngle = (rotationAngle + dt * 180) % 360
                icon.Rotation = rotationAngle
            end)

            if renderConnection then renderConnection:Disconnect() end
            renderConnection = services.RunService.RenderStepped:Connect(function()
                if not isResizing then
                    if renderConnection then
                        renderConnection:Disconnect()
                        renderConnection = nil
                    end
                    return
                end
                local currentPos = services.UserInputService:GetMouseLocation()
                local delta = currentPos - startPos
                local newWidth = math.clamp(startSize.X + delta.X, minWindowSize.X, maxWindowSize.X)
                local newHeight = math.clamp(startSize.Y + delta.Y, minWindowSize.Y, maxWindowSize.Y)
                CloudMain.Size = UDim2.new(0, newWidth, 0, newHeight)
            end)
        end
    end)


    services.UserInputService.InputEnded:Connect(function(input, gpe)
        if gpe then return end
        if isResizing and isPrimaryInput(input) then
            isResizing = false
            if renderConnection then
                renderConnection:Disconnect()
                renderConnection = nil
            end
            if rotationConnection then
                rotationConnection:Disconnect()
                rotationConnection = nil
            end
            Tween(icon, {0.3, "Elastic", "Out"}, {
                Size = UDim2.new(1, -2, 1, -2),
                ImageTransparency = 0.9,
                Rotation = 0
            })
            CloudMain.Draggable = true
            updateLayout()
        end
    end)


    handle.AncestryChanged:Connect(function()
        if isResizing then
            isResizing = false
            if renderConnection then
                renderConnection:Disconnect()
                renderConnection = nil
            end
            if rotationConnection then
                rotationConnection:Disconnect()
                rotationConnection = nil
            end
            CloudMain.Draggable = true
        end
    end)


    handle.MouseEnter:Connect(function()
        if isResizing then return end
        Tween(icon, {0.2, "Quad", "Out"}, {Size = UDim2.new(1, 0, 1, 0)})
    end)
    handle.MouseLeave:Connect(function()
        if isResizing then return end
        Tween(icon, {0.2, "Quad", "Out"}, {Size = UDim2.new(1, -2, 1, -2)})
    end)
end

function section.Slider(_, text, flag, default, min, max, precise, callback)
    callback = callback or function() end
    min = min or 1
    max = max or 10
    default = default or min
    precise = precise or false

    if min > max then
        min, max = max, min
    end
    default = math.clamp(default, min, max)

    library.flags[flag] = default

    assert(text, "No text provided")
    assert(flag, "No flag provided")
    assert(default, "No default value provided")

    local ICONS = {
        half = "rbxassetid://117449275562979",
        off  = "rbxassetid://75742832732503",
        on   = "rbxassetid://136141469398409"
    }

    local SliderModule = Instance.new("Frame")
    SliderModule.Name = "SliderModule"
    SliderModule.Parent = Objs
    SliderModule.BackgroundColor3 = TextColor
    SliderModule.BackgroundTransparency = 1.000
    SliderModule.BorderSizePixel = 0

    local SliderBack = Instance.new("TextButton")
    SliderBack.Name = "SliderBack"
    SliderBack.Parent = SliderModule
    SliderBack.BackgroundColor3 = zyColor
    SliderBack.BackgroundTransparency = interfaceTransparency
    SliderBack.BorderSizePixel = 0

    local BackCorner = Instance.new("UICorner")
    BackCorner.CornerRadius = UDim.new(0, 8)
    BackCorner.Parent = SliderBack

    local Content = Instance.new("Frame")
    Content.Name = "Content"
    Content.Parent = SliderBack
    Content.BackgroundTransparency = 1
    Content.Size = UDim2.new(1, -16, 1, 0)
    Content.Position = UDim2.new(0, 8, 0, 0)

    local Layout = Instance.new("UIListLayout")
    Layout.Parent = Content
    Layout.FillDirection = Enum.FillDirection.Horizontal
    Layout.HorizontalAlignment = Enum.HorizontalAlignment.Left
    Layout.VerticalAlignment = Enum.VerticalAlignment.Center
    Layout.SortOrder = Enum.SortOrder.LayoutOrder
    Layout.Padding = UDim.new(0, 8)

    local Label = Instance.new("TextLabel")
    Label.Name = "Label"
    Label.Parent = Content
    Label.BackgroundTransparency = 1
    Label.Size = UDim2.new(0, 70, 0, 20)
    Label.Font = Enum.Font.Cartoon
    Label.Text = text
    Label.TextColor3 = TextColor
    Label.TextSize = 14
    Label.TextXAlignment = Enum.TextXAlignment.Left
    Label.TextTransparency = NameTransparency
    Label.LayoutOrder = 1

    local LabelGlow = Instance.new("UIStroke")
    LabelGlow.Parent = Label
    LabelGlow.Thickness = 0.5
    LabelGlow.Color = Color3.fromRGB(255, 255, 255)
    LabelGlow.Transparency = 0.1

    local Track = Instance.new("Frame")
    Track.Name = "Track"
    Track.Parent = Content
    Track.BackgroundColor3 = Background
    Track.BorderSizePixel = 0
    Track.Size = UDim2.new(0.4, -40, 0, 8)
    Track.LayoutOrder = 2

    local TrackCorner = Instance.new("UICorner")
    TrackCorner.CornerRadius = UDim.new(1, 0)
    TrackCorner.Parent = Track

    local Fill = Instance.new("Frame")
    Fill.Name = "Fill"
    Fill.Parent = Track
    Fill.BackgroundColor3 = Color3.fromRGB(100, 200, 255)
    Fill.BorderSizePixel = 0
    Fill.Size = UDim2.new(0, 0, 1, 0)

    local FillCorner = Instance.new("UICorner")
    FillCorner.CornerRadius = UDim.new(1, 0)

    IconContainer.BackgroundTransparency = 1

    IconContainer.LayoutOrder = 4

    local function refreshLayout()
        local layout = Content:FindFirstChildOfClass("UIListLayout")
        if layout then
            local old = layout.Padding
            layout.Padding = old + UDim.new(0, 0.001)
            layout.Padding = old
        end
    end
    refreshLayout()

    local Icon = Instance.new("ImageLabel")
    Icon.Name = "StageIcon"
    Icon.Parent = IconContainer
    Icon.BackgroundTransparency = 1
    Icon.Size = UDim2.new(1, 0, 1, 0)
    Icon.Image = ICONS.off
    Icon.ImageTransparency = 0
    Icon.ZIndex = 3

    local function updateValueBoxWidth()
        local textWidth = ValueBox.TextBounds.X + 16
        local maxWidth = 100
        local minWidth = 40
        local newWidth = math.clamp(textWidth, minWidth, maxWidth)
        ValueBG.Size = UDim2.new(0, newWidth, 0, 22)
    end
    ValueBox:GetPropertyChangedSignal("Text"):Connect(updateValueBoxWidth)
    updateValueBoxWidth()

    local function getStage(value)
        if value == 0 then return nil end
        local percent = (value - min) / (max - min)
        percent = math.clamp(percent, 0, 1)
        if percent <= 1/3 then
            return "half"
        elseif percent <= 2/3 then
            return "off"
        else
            return "on"
        end
    end

    local function playIconAnimation(stage)
        if not stage then
            Icon.ImageTransparency = 1
            return
        end
        Icon.Image = ICONS[stage]
        Icon.ImageTransparency = 0
        Tween(Icon, {0.1, "Quad", "Out"}, {
            Size = UDim2.new(1.5, 0, 1.5, 0),
            Rotation = 15,
            ImageTransparency = 0
        })
        Tween(Icon, {0.1, "Quad", "Out"}, {ImageColor3 = Color3.fromRGB(255, 255, 100)})
        task.wait(0.1)
        Tween(Icon, {0.15, "Elastic", "Out"}, {
            Size = UDim2.new(1, 0, 1, 0),
            Rotation = 0,
            ImageColor3 = Color3.fromRGB(255, 255, 255)
        })
    end

    local function updateValue(newValue, skipCallback)
        newValue = math.clamp(newValue, min, max)
        if precise then
            newValue = tonumber(string.format("%.1f", newValue))
        else
            newValue = math.floor(newValue)
        end
        if library.flags[flag] == newValue then return end
        library.flags[flag] = newValue
        ValueBox.Text = tostring(newValue)
        updateValueBoxWidth()
        local percent = (newValue - min) / (max - min)
        Fill.Size = UDim2.new(percent, 0, 1, 0)
        local stage = getStage(newValue)
        playIconAnimation(stage)
        if not skipCallback then
            callback(newValue)
        end
    end

    updateValue(default, true)

    local dragging = false
    local function startDrag(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
            dragging = true
            local pos = services.UserInputService:GetMouseLocation()
            local absPos = Track.AbsolutePosition
            local absSize = Track.AbsoluteSize
            local relativeX = math.clamp(pos.X - absPos.X, 0, absSize.X)
            local newPercent = relativeX / absSize.X
            local newVal = min + (max - min) * newPercent
            updateValue(newVal)
        end
    end

    Track.InputBegan:Connect(startDrag)

    services.UserInputService.InputChanged:Connect(function(input)
        if dragging and (input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch) then
            if not (Track and Track.Parent) then dragging = false; return end
            local pos = services.UserInputService:GetMouseLocation()
            local absPos = Track.AbsolutePosition
            local absSize = Track.AbsoluteSize
            local relativeX = math.clamp(pos.X - absPos.X, 0, absSize.X)
            local newPercent = relativeX / absSize.X
            local newVal = min + (max - min) * newPercent
            updateValue(newVal)
        end
    end)

    services.UserInputService.InputEnded:Connect(function(input)
        if dragging and (input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch) then
            dragging = false
        end
    end)

    local boxFocused = false
    ValueBox.Focused:Connect(function() boxFocused = true end)
    ValueBox.FocusLost:Connect(function()
        boxFocused = false
        local val = tonumber(ValueBox.Text)
        if val then
            updateValue(val)
        else
            ValueBox.Text = tostring(library.flags[flag])
            updateValueBoxWidth()
        end
    end)

    ValueBox:GetPropertyChangedSignal("Text"):Connect(function()
        if not boxFocused then return end
        local filtered = ValueBox.Text:gsub("[^%d%.]", "")
        local parts = filtered:split(".")
        if #parts > 2 then
            filtered = parts[1] .. "." .. parts[2]
        end
        ValueBox.Text = filtered
        local val = tonumber(filtered)
        if val then
            if val > max then
                ValueBox.Text = tostring(max)
            elseif val < min then
                ValueBox.Text = tostring(min)
            end
        end
        updateValueBoxWidth()
    end)

    local funcs = {
        SetValue = function(self, newVal)
            updateValue(newVal)
        end,
        Module = SliderModule
    }
    return funcs
end

                DropdownText.Name = "DropdownText"
                DropdownText.Parent = DropdownTop
                DropdownText.BackgroundColor3 = TextColor
                DropdownText.BackgroundTransparency = 1.000
                DropdownText.BorderSizePixel = 0
                DropdownText.Position = UDim2.new(0.04, 0, 0, 0)
                DropdownText.Size = UDim2.new(0, 120, 0, 30)
                DropdownText.Font = Enum.Font.Cartoon
                DropdownText.PlaceholderColor3 = TextColor
                DropdownText.PlaceholderText = text
                DropdownText.Text = ""
                DropdownText.TextColor3 = TextColor
                DropdownText.TextSize = 14.000
                DropdownText.TextXAlignment = Enum.TextXAlignment.Left

                DropdownModuleL.Name = "DropdownModuleL"
                DropdownModuleL.Parent = DropdownModule
                DropdownModuleL.SortOrder = Enum.SortOrder.LayoutOrder
                DropdownModuleL.Padding = UDim.new(0, 3)

                local setAllVisible = function()
                    local options = DropdownModule:GetChildren()
                    for i = 1, #options do
                        local option = options[i]
                        if option:IsA("TextButton") and option.Name:match("Option_") then
                            option.Visible = true
                        end
                    end
                end

                local searchDropdown = function(text)
                    if not (DropdownModule and DropdownModule.Parent) then return end
                    local options = DropdownModule:GetChildren()
                    for i = 1, #options do
                        local option = options[i]
                        if text == "" then
                            setAllVisible()
                        else
                            if option:IsA("TextButton") and option.Name:match("Option_") then
                                if option.Text:lower():match(text:lower()) then
                                    option.Visible = true
                                else
                                    option.Visible = false
                                end
                            end
                        end
                    end
                end

                local open = false
                local ToggleDropVis = function()
                    if not (DropdownModule and DropdownModule.Parent) then return end
                    open = not open
                    if open then setAllVisible() end
                    DropdownOpen.Image = open and "rbxassetid://111223574026321" or "rbxassetid://137069490345718"
                    local targetSize = UDim2.new(1, 0, 0, (open and DropdownModuleL.AbsoluteContentSize.Y + 4 or 30))
                    Tween(DropdownModule, {0.2, "Back", "Out"}, {Size = targetSize})
                end

                DropdownOpen.MouseButton1Click:Connect(function()
                    PlayClickSound()
                    ToggleDropVis()
                end)

                DropdownText.Focused:Connect(function()
                    if open then return end
                    ToggleDropVis()
                end)

                DropdownText:GetPropertyChangedSignal("Text"):Connect(function()

                    searchDropdown(DropdownText.Text)
                end)

                DropdownModuleL:GetPropertyChangedSignal("AbsoluteContentSize"):Connect(function()

                end)
                DropdownOpen.MouseLeave:Connect(function()
                    Tween(DropdownOpen, {0.2, "Quad", "Out"}, {Size = UDim2.new(0, 20, 0, 20)})
                end)

                local funcs = {}
                funcs.AddOption = function(self, option)
                    local Option = Instance.new("TextButton")
                    local OptionC = Instance.new("UICorner")

                    Option.Name = "Option_" .. option
                    Option.Parent = DropdownModule
                    Option.BackgroundColor3 = zyColor
                    Option.BorderSizePixel = 0
                    Option.Size = UDim2.new(1, 0, 0, 20)

                    OptionC.Parent = Option

                    Option.MouseButton1Click:Connect(function()
                        PlayClickSound()
                        ToggleDropVis()
                        callback(Option.Text)
                        DropdownText.Text = Option.Text
                        library.flags[flag] = Option.Text
                    end)
                end

                funcs.RemoveOption = function(self, option)
                    local option = DropdownModule:FindFirstChild("Option_" .. option)
                    if option then option:Destroy() end
                end

                funcs.SetOptions = function(self, options)
                    for _, v in next, DropdownModule:GetChildren() do
                        if v.Name:match("Option_") then v:Destroy() end
                    end
                    for _, v in next, options do funcs:AddOption(v) end
                end

                funcs:SetOptions(options)
                return funcs
            end
            return section
        end
        return tab
    end
